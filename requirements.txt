-- #######################################################
-- # 1. التنظيف الشامل (مسح كل الجداول المتعلقة بالمشروع)
-- #######################################################

-- يتم استخدام CASCADE لحذف الروابط بين الجداول (الأمر الآمن والوحيد)
DROP TABLE IF EXISTS interactions CASCADE;
DROP TABLE IF EXISTS leads CASCADE;
DROP TABLE IF EXISTS campaigns CASCADE;
DROP TABLE IF EXISTS project_settings CASCADE;
DROP TABLE IF EXISTS users CASCADE; 
DROP TABLE IF EXISTS settings CASCADE; 


-- #######################################################
-- # 2. إعادة الإنشاء الآمن (بناء هيكل Vercel/Next.js)
-- #######################################################

-- 1. جدول المستخدمين (للدخول والأمان)
CREATE TABLE users (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    username TEXT UNIQUE NOT NULL,
    password TEXT NOT NULL,
    role TEXT DEFAULT 'client',
    is_active BOOLEAN DEFAULT TRUE
);

-- 2. جدول الحملات (لتنظيم الـ Hunter)
CREATE TABLE campaigns (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name TEXT NOT NULL,
    niche TEXT,
    location TEXT,
    keywords TEXT[],
    opener TEXT, 
    status TEXT DEFAULT 'draft',
    is_active BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 3. جدول الـ Leads (العملاء) - يتضمن ربط الحملة
CREATE TABLE leads (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    campaign_id UUID REFERENCES campaigns(id) ON DELETE CASCADE,
    phone_number TEXT UNIQUE NOT NULL,
    status TEXT DEFAULT 'new',
    intent_summary TEXT,  
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 4. جدول التفاعلات (الذاكرة - AI Closer)
CREATE TABLE interactions (
    id BIGSERIAL PRIMARY KEY,
    phone_number TEXT NOT NULL,
    message TEXT NOT NULL,
    sender TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    FOREIGN KEY (phone_number) REFERENCES leads(phone_number) ON DELETE CASCADE
);

-- 5. جدول الإعدادات العامة (Settings)
CREATE TABLE settings (
    key TEXT PRIMARY KEY,
    value TEXT
);
